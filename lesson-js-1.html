Введение в JavaScript

JavaScript - это язык программирования, который позволяет создавать интерактивные веб-страницы. В отличие от HTML и CSS, JavaScript является полноценным языком программирования.

Важно!

JavaScript не имеет прямого отношения к Java. Это разные языки с разными возможностями и синтаксисом.

Что можно делать с помощью JavaScript?

  • Добавлять интерактивность на веб-страницы
  • Создавать анимации и динамические эффекты
  • Обрабатывать пользовательские события (клики, ввод данных)
  • Работать с данными на стороне клиента
  • Взаимодействовать с сервером (AJAX)
  • Создавать веб-приложения и игры

Подключение JavaScript к странице

1. Встроенный скрипт

<script>
    // JavaScript код
    console.log('Привет, мир!');
</script>

2. Внешний файл

<script src="script.js"></script>

Рекомендация

Используйте внешние файлы для больших проектов - это упрощает поддержку и позволяет переиспользовать код.

Переменные и типы данных

Переменные используются для хранения данных. В JavaScript есть несколько способов объявления переменных.

Объявление переменных

// var - устаревший способ (не рекомендуется)
var oldVariable = "значение";

// let - современный способ для изменяемых переменных
let counter = 0;

// const - для констант (неизменяемых значений)
const PI = 3.14159;

Рекомендация

Всегда используйте const по умолчанию, и только если переменная будет изменяться - используйте let. Избегайте var.

Типы данных

// Примитивные типы
let string = "Текст";          // Строка
let number = 42;               // Число
let boolean = true;            // Логический тип
let nullValue = null;          // Null
let undefinedValue;            // Undefined
let symbol = Symbol("id");     // Символ (ES6)

// Объектные типы
let object = {                 // Объект
    name: "Иван",
    age: 30
};
let array = [1, 2, 3];         // Массив
let func = function() {        // Функция
    return "Привет";
};

Операторы

Операторы позволяют выполнять операции с переменными и значениями.

Арифметические операторы

let a = 10, b = 3;

console.log(a + b);   // 13 (сложение)
console.log(a - b);   // 7 (вычитание)
console.log(a * b);   // 30 (умножение)
console.log(a / b);   // 3.333... (деление)
console.log(a % b);   // 1 (остаток от деления)
console.log(a ** b);  // 1000 (возведение в степень)
console.log(++a);     // 11 (инкремент)
console.log(--b);     // 2 (декремент)

Операторы сравнения

console.log(5 == "5");    // true (нестрогое равенство)
console.log(5 === "5");   // false (строгое равенство)
console.log(5 != "5");    // false (нестрогое неравенство)
console.log(5 !== "5");   // true (строгое неравенство)
console.log(5 > 3);       // true
console.log(5 < 3);       // false
console.log(5 >= 5);      // true
console.log(5 <= 3);      // false

Логические операторы

console.log(true && false);   // false (И)
console.log(true || false);   // true (ИЛИ)
console.log(!true);           // false (НЕ)

// Короткое замыкание
let result = false && someFunction();  // someFunction не выполнится
let result2 = true || someFunction();  // someFunction не выполнится

Функции

Функции - это блоки кода, которые можно многократно вызывать с разными параметрами.

Объявление функций

// Function Declaration
function greet(name) {
    return "Привет, " + name + "!";
}

// Function Expression
const multiply = function(a, b) {
    return a * b;
};

// Стрелочные функции (ES6)
const divide = (a, b) => {
    return a / b;
};

// Сокращенная форма стрелочных функций
const square = x => x * x;

Параметры и аргументы

// Параметры по умолчанию
function createUser(name, age = 18) {
    return {
        name: name,
        age: age
    };
}

// Rest параметры
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3, 4, 5)); // 15

Замыкания

Замыкания - это функции, которые запоминают свое лексическое окружение.

function createCounter() {
    let count = 0;

    return function() {
        count++;
        return count;
    };
}

const counter = createCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

Объекты и массивы

Объекты и массивы - это сложные структуры данных для хранения коллекций значений.

Объекты

// Создание объекта
const person = {
    firstName: "Иван",
    lastName: "Петров",
    age: 30,
    hobbies: ["чтение", "программирование"],

    // Метод объекта
    getFullName: function() {
        return this.firstName + " " + this.lastName;
    }
};

// Доступ к свойствам
console.log(person.firstName);        // Иван
console.log(person["lastName"]);      // Петров
console.log(person.getFullName());    // Иван Петров

// Добавление свойств
person.email = "ivan@example.com";

// Удаление свойств
delete person.age;

Массивы

// Создание массива
const fruits = ["яблоко", "банан", "апельсин"];

// Доступ к элементам
console.log(fruits[0]);   // яблоко
console.log(fruits[1]);   // банан

// Методы массивов
fruits.push("груша");     // Добавить в конец
fruits.pop();             // Удалить последний
fruits.unshift("киви");   // Добавить в начало
fruits.shift();           // Удалить первый

// Итерация по массиву
fruits.forEach(fruit => {
    console.log(fruit);
});

// Функциональные методы
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(n => n * 2);          // [2, 4, 6, 8, 10]
const even = numbers.filter(n => n % 2 === 0);    // [2, 4]
const sum = numbers.reduce((acc, n) => acc + n, 0); // 15

Работа с DOM

DOM (Document Object Model) - это программный интерфейс для работы с HTML-документами.

Поиск элементов

// Поиск одного элемента
const header = document.getElementById("header");
const firstPara = document.querySelector("p");

// Поиск нескольких элементов
const allDivs = document.getElementsByTagName("div");
const buttons = document.querySelectorAll(".btn");

// Поиск по классу
const elements = document.getElementsByClassName("my-class");

Изменение содержимого

const element = document.getElementById("myElement");

// Изменение текста
element.textContent = "Новый текст";
element.innerText = "Другой текст";

// Изменение HTML
element.innerHTML = "<strong>Жирный текст</strong>";

// Изменение атрибутов
element.setAttribute("class", "new-class");
element.classList.add("active");
element.classList.remove("old-class");
element.classList.toggle("visible");

Создание и удаление элементов

// Создание элемента
const newDiv = document.createElement("div");
newDiv.textContent = "Новый div";

// Добавление в DOM
document.body.appendChild(newDiv);
parentElement.insertBefore(newDiv, referenceElement);

События

События позволяют реагировать на действия пользователя и другие происшествия.

Добавление обработчиков событий

// Способ 1: через атрибут HTML
// <button onclick="handleClick()">Клик</button>

// Способ 2: через свойство
button.onclick = function() {
    console.log("Клик!");
};

// Способ 3: через addEventListener (рекомендуется)
button.addEventListener("click", function(event) {
    console.log("Клик!", event);
});

// Удаление обработчика
button.removeEventListener("click", handlerFunction);

Типы событий

// События мыши
element.addEventListener("click", handler);
element.addEventListener("dblclick", handler);
element.addEventListener("mouseover", handler);
element.addEventListener("mouseout", handler);

// События клавиатуры
document.addEventListener("keydown", handler);
document.addEventListener("keyup", handler);

// События формы
form.addEventListener("submit", handler);
input.addEventListener("change", handler);
input.addEventListener("input", handler);

// Другие события
window.addEventListener("load", handler);
window.addEventListener("resize", handler);
window.addEventListener("scroll", handler);

Объект события

element.addEventListener("click", function(event) {
    console.log(event.type);        // "click"
    console.log(event.target);      // Элемент, на котором произошло событие
    console.log(event.clientX);     // Координата X курсора
    console.log(event.clientY);     // Координата Y курсора
    console.log(event.key);         // Нажатая клавиша (для keyboard events)

    // Предотвращение стандартного поведения
    event.preventDefault();

    // Остановка всплытия
    event.stopPropagation();
});

Асинхронный JavaScript

Асинхронное программирование позволяет выполнять операции без блокировки основного потока.

Callback функции

function loadData(callback) {
    setTimeout(() => {
        const data = { name: "Иван", age: 30 };
        callback(data);
    }, 1000);
}

loadData(function(result) {
    console.log(result); // { name: "Иван", age: 30 }
});

Promises (Обещания)

function loadData() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const success = Math.random() > 0.5;
            if (success) {
                resolve({ name: "Иван", age: 30 });
            } else {
                reject("Ошибка загрузки");
            }
        }, 1000);
    });
}

loadData()
    .then(data => {
        console.log("Данные:", data);
    })
    .catch(error => {
        console.error("Ошибка:", error);
    });

Async/Await

async function loadUserData() {
    try {
        const response = await fetch('https://api.example.com/user');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error("Ошибка:", error);
    }
}

// Использование
loadUserData().then(user => {
    console.log(user);
});

ES6+ возможности

Современный JavaScript (ES6 и новее) предоставляет множество полезных возможностей.

Деструктуризация

// Деструктуризация объектов
const person = { name: "Иван", age: 30, city: "Москва" };
const { name, age } = person;
console.log(name); // Иван
console.log(age);  // 30

// Деструктуризация массивов
const numbers = [1, 2, 3, 4, 5];
const [first, second, ...rest] = numbers;
console.log(first); // 1
console.log(second); // 2
console.log(rest);  // [3, 4, 5]

Spread и Rest операторы

// Spread оператор
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6]

const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };
const merged = { ...obj1, ...obj2 }; // { a: 1, b: 2, c: 3, d: 4 }

// Rest параметры
function sum(...numbers) {
    return numbers.reduce((acc, num) => acc + num, 0);
}

Шаблонные строки

const name = "Иван";
const age = 30;

// Старый способ
const message1 = "Привет, " + name + "! Тебе " + age + " лет.";

// Новый способ (шаблонные строки)
const message2 = `Привет, ${name}! Тебе ${age} лет.`;

// Многострочные строки
const multiline = `
    Это
    многострочная
    строка
`;

Модули

// math.js
export const PI = 3.14159;
export function square(x) {
    return x * x;
}
export default function multiply(a, b) {
    return a * b;
}

// main.js
import multiply, { PI, square } from './math.js';
console.log(multiply(2, 3)); // 6
console.log(PI);             // 3.14159
